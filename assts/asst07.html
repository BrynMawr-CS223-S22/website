3. The answer is 42

Recall the following example from class and Dive into Systems. This mysterious program reliably produces the output 42. In this question, you will trace the assembly to understand why.

#include <stdio.h>

int assign(void) {
    int y = 40;
    return y;
}

int adder(void) {
    int a;
    return a + 2;
}

int main(void) {
    int x;
    assign();
    x = adder();
    printf("x is: %d\n", x);
    return 0;
}
